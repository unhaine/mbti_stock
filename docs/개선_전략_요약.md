#### 1. AI 기반 개인화 🤖

```javascript
// OpenAI API 연동
const generateMBTIExplanation = async (stock, mbti) => {
  const prompt = `
    ${mbti} 성향의 투자자 관점에서
    ${stock.name} 종목을 은유적으로 설명해주세요.
  `

  const response = await openai.chat.completions.create({
    model: 'gpt-4',
    messages: [{ role: 'user', content: prompt }],
  })

  return response.choices[0].message.content
}
```
